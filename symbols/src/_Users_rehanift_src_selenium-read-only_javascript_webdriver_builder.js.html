 18 goog.require('webdriver.WebDriver');
 19 goog.require('webdriver.http.CorsClient');
 20 goog.require('webdriver.http.Executor');
 21 goog.require('webdriver.http.Response');
 22 goog.require('webdriver.node.HttpClient');
 23 goog.require('webdriver.process');
 24 
 25 
 26 /**
 27  * Creates new {@code webdriver.WebDriver} clients.  Upon instantiation, each
 28  * Builder will configure itself based on the following environment variables:
 29  * <dl>
 30  *   <dt>{@code webdriver.Builder.SERVER_URL_ENV}</dt>
 31  *   <dd>Defines the remote WebDriver server that should be used for command
 32  *       command execution; may be overridden using
 33  *       {@code webdriver.Builder.prototype.usingServer}.</dd>
 34  *   <dt>{@code webdriver.Builder.SESSION_ID_ENV}</dt>
 35  *   <dd>Defines the ID of an existing WebDriver session that should be used
 36  *       for new clients. This is most often used for browser-based clients
 37  *       that wish to gain control of the current browser which is already under
 38  *       WebDriver's control.</dd>
 39  * </dl>
 40  * @constructor
 41  */
 42 webdriver.Builder = function() {
 43 
 44   /**
 45    * URL of the remote server to use for new clients; initialized from the
 46    * value of the {@code webdriver.Builder.SERVER_URL_ENV} environment variable,
 47    * but may be overridden using {@code #usingServer(url)}.
 48    * @type {string}
 49    * @private
 50    */
 51   this.serverUrl_ = webdriver.process.getEnv(
 52       webdriver.Builder.SERVER_URL_ENV, webdriver.Builder.DEFAULT_SERVER_URL);
 53 
 54   /**
 55    * ID of an existing WebDriver session that new clients should use.
 56    * Initialized from the value of the {@code webdriver.Builder.SESSION_ID_ENV}
 57    * environment variable, but may be overridden using
 58    * {@code #usingSession(id)}.
 59    * @type {string}
 60    * @private
 61    */
 62   this.sessionId_ =
 63       webdriver.process.getEnv(webdriver.Builder.SESSION_ID_ENV);
 64 
 65   /**
 66    * The desired capabilities to use when creating a new session.
 67    * @type {!Object.<*>}
 68    * @private
 69    */
 70   this.capabilities_ = {};
 71 };
 72 
 73 
 74 /**
 75  * Environment variable that defines the session ID of an existing WebDriver
 76  * session to use when creating clients. If set, all new Builder instances will
 77  * default to creating clients that use this session. To create a new session,
 78  * use {@code #useExistingSession(boolean)}. The use of this environment
 79  * variable requires that {@code webdriver.Builder.SERVER_URL_ENV} also be
 80  * set.
 81  * @type {string}
 82  * @const
 83  * @see webdriver.process.getEnv
 84  */
 85 webdriver.Builder.SESSION_ID_ENV = 'wdsid';
 86 
 87 
 88 /**
 89  * Environment variable that defines the URL of the WebDriver server that
 90  * should be used for all new WebDriver clients. This setting may be overridden
 91  * using {@code #usingServer(url)}.
 92  * @type {string}
 93  * @const
 94  * @see webdriver.process.getEnv
 95  */
 96 webdriver.Builder.SERVER_URL_ENV = 'wdurl';
 97 
 98 
 99 /**
100  * The default URL of the WebDriver server to use if
101  * {@link webdriver.Builder.SERVER_URL_ENV} is not set.
102  * @type {string}
103  * @const
104  */
105 webdriver.Builder.DEFAULT_SERVER_URL = 'http://localhost:4444/wd/hub';
106 
107 
108 /**
109  * Configures which WebDriver server should be used for new sessions. Overrides
110  * the value loaded from the {@code webdriver.Builder.SERVER_URL_ENV} upon
111  * instantion of this instance.
112  * @param {string} url URL of the server to use.
113  * @return {!webdriver.Builder} This Builder instance for chain calling.
114  */
115 webdriver.Builder.prototype.usingServer = function(url) {
116   this.serverUrl_ = url;
117   return this;
118 };
119 
120 
121 /**
122  * Configures the builder to create a client that will use an existing WebDriver
123  * session.
124  * @param {string} id The existing session ID to use.
125  * @return {!webdriver.Builder} This Builder instance for chain calling.
126  */
127 webdriver.Builder.prototype.usingSession = function(id) {
128   this.sessionId_ = id;
129   return this;
130 };
131 
132 
133 /**
134  * Sets the desired capabilities when requesting a new session.
135  * @param {!Object.<*>} capabilities The desired capabilities for a new
136  *     session.
137  * @return {!webdriver.Builder} This Builder instance for chain calling.
138  */
139 webdriver.Builder.prototype.withCapabilities = function(capabilities) {
140   this.capabilities_ = capabilities;
141   return this;
142 };
143 
144 
145 /**
146  * Builds a new {@code webdriver.WebDriver} instance using this builder's
147  * current configuration.
148  * @return {!webdriver.WebDriver} A new WebDriver client.
149  */
150 webdriver.Builder.prototype.build = function() {
151   var executor;
152   if (webdriver.FirefoxDomExecutor.isAvailable()) {
153     executor = new webdriver.FirefoxDomExecutor();
154     return webdriver.WebDriver.createSession(executor, this.capabilities_);
155   } else {
156     var clientCtor = webdriver.process.isNative() ?
157         webdriver.node.HttpClient :
158         webdriver.http.CorsClient;
159 
160     var client = new clientCtor(this.serverUrl_);
161     executor = new webdriver.http.Executor(client);
162 
163     if (this.sessionId_) {
164       return webdriver.WebDriver.attachToSession(executor, this.sessionId_);
165     } else if (webdriver.process.isNative()) {
166       return webdriver.WebDriver.createSession(executor, this.capabilities_);
167     } else {
168       throw new Error('Unable to create a new client for this browser. The ' +
169           'WebDriver session ID has not been defined.');
170     }
171   }
172 };
