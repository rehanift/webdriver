 15 goog.provide('webdriver.Session');
 16 
 17 
 18 /**
 19  * Contains information about a WebDriver session.
 20  * @param {string} id The session ID.
 21  * @param {!Object.<*>} capabilities A map describing the capabilities
 22  *     of this session.
 23  * @constructor
 24  */
 25 webdriver.Session = function(id, capabilities) {
 26 
 27   /**
 28    * The session ID.
 29    * @type {string}
 30    */
 31   this.id = id;
 32 
 33   /**
 34    * A map describing the capabilities of this session.
 35    * @type {!Object.<*>}
 36    */
 37   this.capabilities = capabilities;
 38 };
 39 
 40 
 41 /**
 42  * @return {string} This session's ID.
 43  */
 44 webdriver.Session.prototype.getId = function() {
 45   return this.id;
 46 };
 47 
 48 
 49 /**
 50  * @return {!Object.<*>} This session's capabilities.
 51  */
 52 webdriver.Session.prototype.getCapabilities = function() {
 53   return this.capabilities;
 54 };
 55 
 56 
 57 /**
 58  * Retrieves the value of a specific capability.
 59  * @param {string} key The capability to retrieve.
 60  * @return {*} The capability value.
 61  */
 62 webdriver.Session.prototype.getCapability = function(key) {
 63   return this.capabilities[key];
 64 };
 65 
 66 
 67 /**
 68  * Returns the JSON representation of this object, which is just the string
 69  * session ID.
 70  * @return {string} The JSON representation of this Session.
 71  */
 72 webdriver.Session.prototype.toJSON = function() {
 73   return this.id;
 74 };
